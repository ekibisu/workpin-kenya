1.  What We Found — Current Schema Audit
The current WorkPin database has 10 tables. Some are well-structured and can be kept with minor changes. Others need significant reworking to support the WorkPin Kenya communication model, M-Pesa payments, and the full WorkThread lifecycle.

3.  Migration SQL Scripts
Run these scripts in order in the Supabase SQL Editor. Each script is safe to run independently. Do not skip steps — later scripts depend on earlier ones.

Run each block in the Supabase Dashboard → SQL Editor. After each block, verify the changes in Table Editor before proceeding.

STEP 1 — Add role to profiles & backfill from user_roles
-- Step 1a: Add role column to profiles
ALTER TABLE profiles
  ADD COLUMN IF NOT EXISTS role TEXT NOT NULL DEFAULT 'client'
    CHECK (role IN ('client', 'provider'));

ALTER TABLE profiles
  ADD COLUMN IF NOT EXISTS onboarding_complete BOOLEAN NOT NULL DEFAULT false;

-- Step 1b: Backfill role from user_roles
-- (converts 'customer' -> 'client', 'provider' stays 'provider')
UPDATE profiles p
SET role = CASE ur.role
             WHEN 'customer' THEN 'client'
             WHEN 'provider' THEN 'provider'
             ELSE 'client'
           END
FROM user_roles ur
WHERE ur.user_id = p.id;

-- Step 1c: Verify backfill before dropping
-- SELECT id, full_name, role FROM profiles;
-- (Expected: 2 rows role='client', 2 rows role='provider')

-- Step 1d: Drop user_roles (only after verifying step 1c)
DROP TABLE IF EXISTS user_roles;

STEP 6 — Create work_threads (before migrating messages or quotes)
CREATE TABLE IF NOT EXISTS work_threads (
  id              UUID   PRIMARY KEY DEFAULT gen_random_uuid(),
  job_request_id  UUID   REFERENCES job_requests(id) ON DELETE SET NULL,
  client_id       UUID   NOT NULL REFERENCES profiles(id),
  provider_id     UUID   NOT NULL REFERENCES profiles(id),
  status          TEXT   NOT NULL DEFAULT 'new'
                           CHECK (status IN ('new','quoted','negotiating','active',
                                             'completed','reviewed','disputed')),
  created_at      TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- Enable RLS: only participants can see their threads
ALTER TABLE work_threads ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Participants can view their threads"
  ON work_threads FOR SELECT
  USING (client_id = auth.uid() OR provider_id = auth.uid());

CREATE POLICY "Authenticated users can create threads"
  ON work_threads FOR INSERT
  WITH CHECK (auth.uid() IS NOT NULL);

CREATE POLICY "Participants can update their threads"
  ON work_threads FOR UPDATE
  USING (client_id = auth.uid() OR provider_id = auth.uid());

-- Enable Realtime on work_threads
ALTER PUBLICATION supabase_realtime ADD TABLE work_threads;

-- Create one work_thread per unique job_request that had direct messages.
-- We use the customer_id from job_requests as client_id.
-- We derive provider_id from the first quote on that request.
INSERT INTO work_threads (job_request_id, client_id, provider_id, status)
SELECT DISTINCT ON (jr.id)
  jr.id         AS job_request_id,
  jr.client_id  AS client_id,
  q.provider_id AS provider_id,
  CASE jr.status
    WHEN 'completed' THEN 'completed'
    WHEN 'pending'   THEN 'negotiating'
    ELSE 'quoted'
  END           AS status
FROM job_requests jr
JOIN quotes q ON q.request_id = jr.id
WHERE jr.id IN (SELECT DISTINCT request_id FROM direct_messages)
ORDER BY jr.id, q.created_at ASC;

STEP 7 — Alter quotes table
-- Add work_thread_id FK
ALTER TABLE quotes
  ADD COLUMN IF NOT EXISTS work_thread_id UUID REFERENCES work_threads(id),
  ADD COLUMN IF NOT EXISTS timeline        TEXT;

-- Rename price → price_kes
ALTER TABLE quotes RENAME COLUMN price TO price_kes;

-- Backfill work_thread_id from work_threads
UPDATE quotes q
SET work_thread_id = wt.id
FROM work_threads wt
WHERE wt.job_request_id = q.request_id;

-- Rename 'rejected' status values to 'declined'
UPDATE quotes SET status = 'declined' WHERE status = 'rejected';

-- Update status CHECK constraint
-- First find and drop old constraint, then add new one:
ALTER TABLE quotes
  ADD CONSTRAINT quotes_status_check_v2
    CHECK (status IN ('pending','accepted','declined','countered'));

STEP 8 — Migrate direct_messages → messages + drop conversation_read_status
-- Step 8a: Alter the existing empty messages table to match new schema
ALTER TABLE messages
  ADD COLUMN IF NOT EXISTS work_thread_id UUID REFERENCES work_threads(id),
  ADD COLUMN IF NOT EXISTS type           TEXT NOT NULL DEFAULT 'text'
                             CHECK (type IN ('text', 'quote', 'system')),
  ADD COLUMN IF NOT EXISTS read_at        TIMESTAMPTZ;

-- Rename content → body if column name differs
DO $$
BEGIN
  IF EXISTS (SELECT 1 FROM information_schema.columns
             WHERE table_name='messages' AND column_name='content') THEN
    ALTER TABLE messages RENAME COLUMN content TO body;
  END IF;
  IF NOT EXISTS (SELECT 1 FROM information_schema.columns
                 WHERE table_name='messages' AND column_name='body') THEN
    ALTER TABLE messages ADD COLUMN body TEXT NOT NULL DEFAULT '';
  END IF;
END $$;

-- Step 8b: Migrate data from direct_messages into messages
-- Map request_id → work_thread_id via work_threads table
INSERT INTO messages (id, work_thread_id, sender_id, body, type, created_at)
SELECT
  dm.id,
  wt.id           AS work_thread_id,
  dm.sender_id,
  dm.content      AS body,
  'text'          AS type,
  dm.created_at
FROM direct_messages dm
JOIN work_threads wt ON wt.job_request_id = dm.request_id;

-- Step 8c: Enable Realtime on messages
ALTER PUBLICATION supabase_realtime ADD TABLE messages;

-- Step 8d: Enable RLS on messages
ALTER TABLE messages ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Thread participants can view messages"
  ON messages FOR SELECT
  USING (
    EXISTS (
      SELECT 1 FROM work_threads wt
      WHERE wt.id = work_thread_id
        AND (wt.client_id = auth.uid() OR wt.provider_id = auth.uid())
    )
  );

CREATE POLICY "Thread participants can insert messages"
  ON messages FOR INSERT
  WITH CHECK (
    auth.uid() = sender_id AND
    EXISTS (
      SELECT 1 FROM work_threads wt
      WHERE wt.id = work_thread_id
        AND (wt.client_id = auth.uid() OR wt.provider_id = auth.uid())
    )
  );

-- Step 8e: Drop old tables
DROP TABLE IF EXISTS direct_messages;
DROP TABLE IF EXISTS conversation_read_status;

STEP 10 — Create all new tables
-- bookings
CREATE TABLE IF NOT EXISTS bookings (
  id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  work_thread_id   UUID NOT NULL REFERENCES work_threads(id),
  scheduled_at     TIMESTAMPTZ NOT NULL,
  duration_minutes INTEGER,
  is_quick_pin     BOOLEAN NOT NULL DEFAULT false,
  created_at       TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- disputes
CREATE TABLE IF NOT EXISTS disputes (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  work_thread_id UUID NOT NULL REFERENCES work_threads(id),
  filed_by_id    UUID NOT NULL REFERENCES profiles(id),
  reason         TEXT NOT NULL,
  description    TEXT,
  evidence_urls  TEXT[] DEFAULT '{}',
  status         TEXT NOT NULL DEFAULT 'open'
                   CHECK (status IN ('open', 'in_review', 'resolved')),
  created_at     TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- provider_templates
CREATE TABLE IF NOT EXISTS provider_templates (
  id          UUID    PRIMARY KEY DEFAULT gen_random_uuid(),
  provider_id UUID    NOT NULL REFERENCES profiles(id) ON DELETE CASCADE,
  title       TEXT    NOT NULL,
  body        TEXT    NOT NULL CHECK (char_length(body) <= 500),
  sort_order  INTEGER NOT NULL DEFAULT 0,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- provider_wallets
CREATE TABLE IF NOT EXISTS provider_wallets (
  id                   UUID    PRIMARY KEY DEFAULT gen_random_uuid(),
  provider_id          UUID    NOT NULL UNIQUE REFERENCES profiles(id) ON DELETE CASCADE,
  available_balance_kes INTEGER NOT NULL DEFAULT 0,
  pending_balance_kes   INTEGER NOT NULL DEFAULT 0,
  updated_at           TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- wallet_transactions
CREATE TABLE IF NOT EXISTS wallet_transactions (
  id             UUID    PRIMARY KEY DEFAULT gen_random_uuid(),
  provider_id    UUID    NOT NULL REFERENCES profiles(id),
  type           TEXT    NOT NULL CHECK (type IN ('credit', 'debit', 'fee')),
  amount_kes     INTEGER NOT NULL,
  description    TEXT,

  created_at     TIMESTAMPTZ NOT NULL DEFAULT now()
);

-- fixed_price_services (Quick Pins)

  duration_minutes    INTEGER,
  availability_json   JSONB   NOT NULL DEFAULT '{}',

-- Create spatial index for fast radius queries

  ON provider_profiles USING GIST (geom);

-- Create the provider matching function used by the job request flow
CREATE OR REPLACE FUNCTION match_providers(
  job_lat    DOUBLE PRECISION,
  job_lng    DOUBLE PRECISION,
  job_cat    TEXT,
  max_results INTEGER DEFAULT 8
)
RETURNS TABLE (
  provider_id UUID,
  distance_km DOUBLE PRECISION,
  score       DOUBLE PRECISION
) AS $$
BEGIN
  RETURN QUERY
  SELECT
    pp.user_id                                    AS provider_id,
    ST_Distance(
      pp.geom::geography,
      ST_SetSRID(ST_MakePoint(job_lng, job_lat), 4326)::geography
    ) / 1000.0                                    AS distance_km,
    (
      COALESCE(pp.avg_rating, 0) * 0.4 +
      (1.0 / NULLIF(pp.response_time_minutes, 0)) * 0.3 * 100 +
      (1.0 / NULLIF(
        ST_Distance(
          pp.geom::geography,
          ST_SetSRID(ST_MakePoint(job_lng, job_lat), 4326)::geography
        ) / 1000.0, 0
      )) * 0.3
    )                                             AS score
  FROM provider_profiles pp
  WHERE
    job_cat = ANY(pp.categories)
    AND pp.geom IS NOT NULL
    AND ST_DWithin(
      pp.geom::geography,
      ST_SetSRID(ST_MakePoint(job_lng, job_lat), 4326)::geography,
      pp.service_radius_km * 1000
    )
  ORDER BY score DESC
  LIMIT max_results;
END;
$$ LANGUAGE plpgsql;
